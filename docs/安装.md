# 安装

```bash
# 进入ssh，创建目录
mkdir ~/code
mkdir ~/download
cd ~

# 只升级所有包，不升级软件和系统内核
yum -y upgrade
# 安装axel工具，该工具支持多路下载和断点续传
yum -y install axel
# 安装编译redis的工具
yum -y install gcc automake autoconf libtool make

# 安装nvm
wget -qO- https://raw.githubusercontent.com/nvm-sh/nvm/v0.34.0/install.sh | bash
# 重新进入下 ssh
nvm -v

# 使用nvm安装node
nvm install --lts
# 查看版本
node -v
npm -v

# 全局安装cnpm
npm install -g cnpm --registry=https://registry.npm.taobao.org

```

## 安装 mongo

```bash
# 安装mongo
# https://www.mongodb.com/download-center/community
# 下载mongo
cd ~/download
axel https://fastdl.mongodb.org/linux/mongodb-linux-x86_64-4.0.9.tgz
# 解压
tar -xvzf mongodb-linux-x86_64-4.0.9.tgz
# 移动到 /usr/local
mv mongodb-linux-x86_64-4.0.9 /usr/local/mongodb
# 进入 /usr/local/mongodb
cd /usr/local/mongodb
# 配置环境变量
vim /etc/profile
# 最后一行加入
export PATH=$PATH:/usr/local/mongodb/bin
# 生效配置
source /etc/profile
# 创建日志和数据目录
mkdir /data/mongodb
mkdir /var/log/mongodb
touch /var/log/mongodb/mongodb.log
# 创建并编辑配置文件
touch /usr/local/mongodb/mongodb.conf
vim /usr/local/mongodb/mongodb.conf
# 根据上边路径填写内容
port= 27017 #端口  
dbpath= /data/mongodb #数据库存文件存放目录  
logpath= /var/log/mongodb/mongodb.log #日志文件存放路径  
logappend= true #使用追加的方式写日志  
# 保存，启动命令
mongod --config /usr/local/mongodb/mongodb.conf &
# 查看mongo运行情况
ps -ef | grep mongo
# 查看mongo运行日志
tail -f /var/log/mongodb/mongodb.log
```

## 安装redis

```bash
# 安装redis
# https://redis.io/download
# 下载redis
cd ~/download
axel http://download.redis.io/releases/redis-5.0.4.tar.gz
# 解压
tar xzf redis-5.0.4.tar.gz
# 移动
mv redis-5.0.4 /usr/local/redis
# 编译
cd /usr/local/redis
make && make install
# 出错后清空辨识失败残留
make distclean
# 配置环境变量
vim /etc/profile
# 最后一行接入
export PATH=$PATH:/usr/local/redis/src
# 生效配置
source /etc/profile
# 启动redis
/usr/local/redis/src/redis-server &
# 查看redis运行情况
ps -ef | grep redis
```
